Request Example

The following sample request includes three types of field objects.

{

"group_by": [ "network_no" ],

"fields": [ "network_no", "rating", "share",

{

"demo_no": 652,

"alias": "Rating A25-54",

"field": "rating"

},

{

"demo_no": 652,

"alias": "Share A25-54",

"field": "share"

}

],

"filters": [

[ "range", "day", "2018-10-01", "2018-10-28"],

[ "in", "network_no", 3, 4, 5 ]

}
